DIST_1 (x, y) :
    n <- x.taille + 1
    m <- y.taille + 1
    T <- entier[n][m]

    pour i allant de 0 a n faire :
        pour j allant de 0 a m faire  :

                T[i][j] <- j*2

            sinon si j==0 faire:
                T[i][j] <- i*2
